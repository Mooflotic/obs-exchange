+/**
+ * Esito massa archivia rumore (sempre «Archiviate N», anche se 0 saltate).
+ * @param {{ rejected?: number, skipped_non_pending?: number, not_found?: number }} res
+ * @returns {string}
+ */
+export function formatArchiveNotice(res) {
+  const rejected = Number(res?.rejected || 0);
+  const skipped = Number(res?.skipped_non_pending || 0);
+  const missing = Number(res?.not_found || 0);
+  let msg = `Archiviate ${rejected}`;
+  if (skipped) msg += `, saltate non-pending ${skipped}`;
+  if (missing) msg += `, non trovate ${missing}`;
+  return msg;
+}
+
+/**
+ * Righe rumore senza proposal_id (escluse dalla massa).
+ * @param {object[]} rumoreRows
+ * @param {object[]} massEligibleRows
+ */
+export function rumoreMissingProposalCount(rumoreRows, massEligibleRows) {
+  const rumoreN = (rumoreRows || []).length;
+  const eligibleN = (massEligibleRows || []).length;
+  return Math.max(0, rumoreN - eligibleN);
+}
