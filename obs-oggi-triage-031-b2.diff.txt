+@router.post("/{asset_id}/reject-name-proposal", response_model=AssetOut)
+def reject_name_proposal(
+    asset_id: int,
+    request: Request,
+    source: str = Query(...),
+    proposal_id: Optional[int] = Query(None),
+    db: Session = Depends(get_db),
+    settings: Settings = Depends(get_settings),
+    user: User = Depends(require_role("operator")),
+):
+    """D6: archivia NameProposal come rejected. Non tocca assets.status / name."""
+    check_csrf(request, settings)
+    asset = db.get(Asset, asset_id)
+    if not asset:
+        raise HTTPException(404, "Asset non trovato")
+    candidates = [p for p in asset.name_proposals if p.source == source]
+    if proposal_id is not None:
+        candidates = [p for p in candidates if p.id == proposal_id]
+    if not candidates:
+        raise HTTPException(404, "Proposta non trovata")
+    pending = [p for p in candidates if (p.status or "") == "pending"]
+    if not pending:
+        raise HTTPException(409, "Nessuna proposta pending da rifiutare")
+    prop = sorted(
+        pending,
+        key=lambda item: item.updated_at or item.created_at,
+        reverse=True,
+    )[0]
+    prev_status = prop.status or "pending"
+    prop.status = "rejected"
+    prop.status_reason = "rejected_from_oggi"
+    prop.updated_at = datetime.utcnow()
+    audit(
+        db,
+        user=user,
+        action="asset.reject_name_proposal",
+        entity="asset",
+        entity_id=str(asset.id),
+        detail={
+            "proposal_id": prop.id,
+            "source": source,
+            "value": prop.value,
+            "previous_status": prev_status,
+        },
+    )
+    db.commit()
+    db.refresh(asset)
+    return _serialize(db, asset)

+@proposals_router.post("/reject-bulk")
+def reject_name_proposals_bulk(
+    body: RejectNameBulkIn,
+    request: Request,
+    db: Session = Depends(get_db),
+    settings: Settings = Depends(get_settings),
+    user: User = Depends(require_role("operator")),
+):
+    """D2: rifiuta in massa NameProposal pending. ids = PK name_proposals."""
+    check_csrf(request, settings)
+    ids = [int(x) for x in (body.ids or []) if x is not None]
+    rejected = 0
+    skipped_non_pending = 0
+    not_found = 0
+    seen: set[int] = set()
+    for pid in ids:
+        if pid in seen:
+            continue
+        seen.add(pid)
+        prop = db.get(NameProposal, pid)
+        if not prop:
+            not_found += 1
+            continue
+        if (prop.status or "") != "pending":
+            skipped_non_pending += 1
+            continue
+        prop.status = "rejected"
+        prop.status_reason = "rejected_bulk_oggi"
+        prop.updated_at = datetime.utcnow()
+        rejected += 1
+    if rejected:
+        audit(
+            db,
+            user=user,
+            action="name_proposal.reject_bulk",
+            entity="name_proposal",
+            entity_id="",
+            detail={
+                "requested": len(seen),
+                "rejected": rejected,
+                "skipped_non_pending": skipped_non_pending,
+                "not_found": not_found,
+            },
+        )
+        db.commit()
+    return {
+        "rejected": rejected,
+        "skipped_non_pending": skipped_non_pending,
+        "not_found": not_found,
+        "requested": len(seen),
+    }

+/** Grouping helpers for Oggi name-proposal triage (OBS-031 B2). */
+
+/**
+ * @param {Array<object>} rows from buildTriageRows
+ * @returns {{ adotta: object[], verifica: object[], rumore: object[] }}
+ */
+export function groupTriageRows(rows) {
+  const adotta = [];
+  const verifica = [];
+  const rumore = [];
+  for (const r of rows || []) {
+    if (r.recommended_action === "adotta") adotta.push(r);
+    else if (r.recommended_action === "archivia") rumore.push(r);
+    else verifica.push(r);
+  }
+  return { adotta, verifica, rumore };
+}
+
+/** @param {object} row */
+export function displayCurrentName(row) {
+  const n = row?.current_name;
+  if (n == null || String(n).trim() === "") return "anonimo";
+  return String(n).trim();
+}
+
+/** @param {object} row */
+export function displayMac(row) {
+  const macs = row?.macs || [];
+  return macs[0] || "—";
+}
+
+/** @param {object} row */
+export function displayIp(row) {
+  const ips = row?.ips || [];
+  return ips[0] || "—";
+}
+
+/** @param {object} row */
+export function displayPort(row) {
+  const p = row?.port_label;
+  if (p == null || String(p).trim() === "") return "—";
+  return String(p).trim();
+}
+
+/** @param {number} confidence 0..1 */
+export function formatConfidencePct(confidence) {
+  const c = Number(confidence || 0);
+  const pct = c <= 1 ? Math.round(c * 100) : Math.round(c);
+  return `${pct}%`;
+}
+
+/** @param {string} verdict */
+export function verdictTone(verdict) {
+  if (verdict === "upgrade" || verdict === "adotta") return "ok";
+  if (verdict === "collide" || verdict === "chassis") return "attn";
+  if (verdict === "rumore") return "muted";
+  return "muted";
+}
+
+/** @param {string} verdict */
+export function verdictLabel(verdict) {
+  if (verdict === "upgrade") return "upgrade";
+  if (verdict === "collide") return "collide";
+  if (verdict === "chassis") return "chassis";
+  if (verdict === "rumore") return "rumore";
+  return verdict || "—";
+}
