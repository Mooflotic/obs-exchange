+/**
+ * D9: sotto-gruppi Verifica nell'ordine collide → chassis → manual-upgrade → sotto-soglia.
+ * @param {object[]} verificaRows
+ * @returns {{
+ *   collide: object[],
+ *   chassis: object[],
+ *   manualUpgrade: object[],
+ *   sottoSoglia: object[],
+ * }}
+ */
+export function splitVerifica(verificaRows) {
+  const collide = [];
+  const chassis = [];
+  const manualUpgrade = [];
+  const sottoSoglia = [];
+  for (const r of verificaRows || []) {
+    if (r.verdict === "collide") collide.push(r);
+    else if (r.verdict === "chassis") chassis.push(r);
+    else if (r.verdict === "upgrade" && r.manual_name) manualUpgrade.push(r);
+    else sottoSoglia.push(r);
+  }
+  return { collide, chassis, manualUpgrade, sottoSoglia };
+}

+/**
+ * D10: id di TUTTE le NameProposal pending con score(proposta) ≤ score(nome attuale).
+ * Esclude chassis (D5) e status=ignorato. Nessuna API.
+ *
+ * @param {Array<object>} assets
+ * @param {object} [chassisPayload]
+ * @returns {number[]}
+ */
+export function noiseProposalIds(assets, chassisPayload = {}) {
+  const list = Array.isArray(assets) ? assets : [];
+  const chassisIds = chassisMemberIds(chassisPayload);
+  const ids = [];
+  for (const asset of list) {
+    if ((asset?.status || "") === "ignorato") continue;
+    if (chassisIds.has(Number(asset?.id))) continue;
+    const currentName = String(asset?.name || "").trim();
+    const scoreCurrent = scoreSpecificity(currentName);
+    const proposals = (asset?.proposals || asset?.name_proposals || []).filter(
+      (p) => (p.status || "pending") === "pending" && p.value && p.id != null,
+    );
+    for (const p of proposals) {
+      const proposedName = String(p.value || "").trim();
+      if (!proposedName) continue;
+      if (scoreSpecificity(proposedName) <= scoreCurrent) {
+        ids.push(Number(p.id));
+      }
+    }
+  }
+  return ids;
+}
